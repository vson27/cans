//UAF

#include <stdio.h>
#include <stdlib.h>

int main() {
    int *ptr = (int*)malloc(sizeof(int));
    *ptr = 100;
    printf("Value before free: %d\n", *ptr);
    free(ptr);
    printf("Memory freed\n");
    printf("Access after free: %d\n", *ptr);  // Vulnerability
    return 0;
}

    free(ptr);
    ptr = NULL;
    if(ptr != NULL)
        printf("%d\n", *ptr);
    else

}
