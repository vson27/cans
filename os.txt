GEN RSA 
openssl genrsa -out private.key 2048
openssl rsa -in private.key -pubout -out public.key
(enc) openssl rsautl -encrypt -inkey public.key -pubin -in message.txt -out encrypted.bin 
(dec) openssl rsautl -decrypt -inkey private.key -in encrypted.bin -out decrypted.txt
or
openssl genpkey -algorithm RSA -out rsa_key.pem -pkeyopt rsa_keygen_bits:2048
openssl pkey -in rsa_key.pem -text -noout
openssl pkey -in rsa_key.pem -pubout -out rsa_public.pem
echo "Hello CNS Lab" > message.txt
openssl pkeyutl \
  -encrypt \
  -pubin \
  -inkey rsa_public.pem \
  -in message.txt \
  -out encrypted.bin
openssl pkeyutl \
  -decrypt \
  -inkey rsa_key.pem \
  -in encrypted.bin \
  -out decrypted.txt
cat decrypted.txt

RSA SIGN OR DIG SIG
openssl dgst -sha256 -sign private.key -out sign.bin message.txt 
(ver) openssl dgst -sha256 -verify public.key -signature sign.bin message.txt

HMAC GEN
openssl dgst -sha256 -hmac "secretkey" message.txt

SYM AES
openssl enc -aes-256-cbc -salt -in message.txt -out cipher.bin
openssl enc -aes-256-cbc -d -in cipher.bin -out recovered.txt
